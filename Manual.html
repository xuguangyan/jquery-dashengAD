第一步、引用JS文件及CSS文件

<link rel="stylesheet" type="text/css" href="css/dashengAD.css" />
<script type="text/javascript" src="js/jquery-1.3.2.min.js"></script>
<script type="text/javascript" src="js/jquery.dashengAD.js"></script>

第二步、插入JS代码

<script language="javascript">
$(document).ready(function(){
  $("#ds_box").dashengAD({
    way: "rollh",//显示方式:show|fade|slide|rollh|rollv 默认值:fade
    speed: "slow", //显示速度:"fast"|"nomarl"|"slow"|(毫秒数) 默认值:"nomarl"
    loop:true, //无缝轮滚:true|false 默认值:false
    auto:true, //自动滚动:true|false 默认值:false
    delay:2 //自滚延时:(秒) 默认值:3
  });
});
</script>
